/**
 * Function to determine if number is even
 * 
 * @function isEven
 * 
 * @example
 * 
 * var a = 1;
 * var b = 2;
 * 
 * var sum = add(1, 2);
 * 
 * console.log(sum);
 * 
 * @param {number} a - The number you want to check
 * 
 * @returns {boolean} Boolean indicating whether value is even
*/
function isEven(a){
    return a % 2 === 0;
}
